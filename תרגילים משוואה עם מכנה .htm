<script>

document.addEventListener('DOMContentLoaded', function() {

const exercisesContainer = document.getElementById('exercises');

const exercises = [

{ equation: 'x/3 = 9', answer: '27' },

{ equation: 'x/4 = 8', answer: '32' },

{ equation: 'x/5 = 10', answer: '50' },

{ equation: 'x/6 = 12', answer: '72' },

{ equation: 'x/7 = 14', answer: '98' },

// ניתן להוסיף עוד משוואות לפי הצורך

];

 

exercises.forEach((exercise, index) => {

const exerciseHTML = `

<div class="exercise">

<p>תרגיל ${index + 1}: פתור את המשוואה ${exercise.equation}</p>

<input type="text" id="answer${index}" placeholder="הקלד את תשובתך כאן">

<button onclick="checkAnswer(${index}, '${exercise.answer}')">בדוק תשובה</button>

<p id="result${index}"></p>

</div>

`;

exercisesContainer.innerHTML += exerciseHTML;

});

});

 

function checkAnswer(index, correctAnswer) {

const userAnswer = document.getElementById(`answer${index}`).value;

const resultDisplay = document.getElementById(`result${index}`);

if (userAnswer === correctAnswer) {

resultDisplay.innerText = "נכון!";

} else {

resultDisplay.innerText = "לא נכון, נסה שוב.";

}

}
